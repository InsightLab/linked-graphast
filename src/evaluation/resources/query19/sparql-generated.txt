SELECT  ?Movie ?Actor ?Actress ?Movie_title ?Actress_birthName
WHERE
  { FILTER regex(lcase(str(?Movie_title)), "^finding nemo$", "i")
    ?Movie    a                     <http://www.movieontology.org/2009/10/01/movieontology.owl#Movie> .
    ?Actor    a                     <http://dbpedia.org/ontology/Actor> .
    ?Movie    <http://www.movieontology.org/2009/10/01/movieontology.owl#hasMaleActor>  ?Actor .
    ?Actress  a                     <http://www.movieontology.org/2009/11/09/movieontology.owl#Actress> .
    ?Actor    <http://www.movieontology.org/2009/10/01/movieontology.owl#hasActress>  ?Actress .
    ?Movie    <http://www.movieontology.org/2009/10/01/movieontology.owl#title>  ?Movie_title .
    ?Actress  <http://dbpedia.org/ontology/birthName>  ?Actress_birthName
  }

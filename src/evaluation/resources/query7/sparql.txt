PREFIX : <http://www.movieontology.org/2009/11/09/movieontology.owl#>
PREFIX mo: <http://www.movieontology.org/2009/10/01/movieontology.owl#>
PREFIX dbpedia: <http://dbpedia.org/ontology/>

SELECT ?title
WHERE {
      ?movie a mo:Movie .
      ?actor1 a dbpedia:Actor .
      ?actor1 dbpedia:birthName "Brando, Marlon" .
      ?movie mo:hasMaleActor ?actor1 .
      ?actor2 a dbpedia:Actor .
      ?actor2 dbpedia:birthName "Pacino, Al" .
      ?movie mo:hasMaleActor ?actor2 .
      ?movie mo:title ?title
}

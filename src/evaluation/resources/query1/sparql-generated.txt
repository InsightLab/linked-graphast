    ?TVSeries  a                    <http://www.movieontology.org/2009/10/01/movieontology.owl#TVSeries> .
    ?Actor    a                     <http://dbpedia.org/ontology/Actor> .
    ?TVSeries  <http://www.movieontology.org/2009/10/01/movieontology.owl#hasMaleActor>  ?Actor ;
              <http://www.movieontology.org/2009/10/01/movieontology.owl#title>  ?TVSeries_title .
    ?Actor    <http://dbpedia.org/ontology/birthName>  ?Actor_birthName
  }

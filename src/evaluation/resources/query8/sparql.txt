PREFIX : <http://www.movieontology.org/2009/11/09/movieontology.owl#>
PREFIX mo: <http://www.movieontology.org/2009/10/01/movieontology.owl#>
PREFIX dbpedia: <http://dbpedia.org/ontology/>

SELECT ?title ?rating
WHERE {
    ?movie a mo:Movie; mo:title ?title; mo:isProducedBy ?prod .
    ?prod :hasCompanyLocation [ a mo:Europe ] .
    ?movie mo:imdbrating ?rating .
}

PREFIX : <http://www.movieontology.org/2009/11/09/movieontology.owl#>
PREFIX mo: <http://www.movieontology.org/2009/10/01/movieontology.owl#>
PREFIX dbpedia: <http://dbpedia.org/ontology/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>

SELECT ?title ?rating
WHERE {
      ?movie a mo:Movie .
      ?actor a dbpedia:Actor .
      ?actor dbpedia:birthName "Willis, Bruce"^^xsd:string .
      ?movie mo:hasMaleActor ?actor .
      ?movie mo:imdbrating ?rating .
      ?movie mo:title ?title
}

SELECT  ?Movie ?Genre ?Editor ?Movie_productionStartYear ?Movie_title ?Movie_imdbrating ?Movie_gross ?Actor ?Actor_birthName ?Editor_birthName ?Producer ?Producer_birthName ?Film_Director ?Film_Director_birthName ?Movie_budget
WHERE
  { FILTER regex(lcase(str(?Movie_title)), "^finding nemo$", "i")
    ?Movie    a                     <http://www.movieontology.org/2009/10/01/movieontology.owl#Movie> .
    ?Genre    a                     <http://www.movieontology.org/2009/10/01/movieontology.owl#Genre> .
    ?Movie    <http://www.movieontology.org/2009/10/01/movieontology.owl#belongsToGenre>  ?Genre .
    ?Editor   a                     <http://www.movieontology.org/2009/10/01/movieontology.owl#Editor> .
    ?Movie    <http://www.movieontology.org/2009/10/01/movieontology.owl#hasEditor>  ?Editor ;
              <http://dbpedia.org/ontology/productionStartYear>  ?Movie_productionStartYear ;
              <http://www.movieontology.org/2009/10/01/movieontology.owl#title>  ?Movie_title ;
              <http://www.movieontology.org/2009/10/01/movieontology.owl#imdbrating>  ?Movie_imdbrating ;
              <http://dbpedia.org/ontology/gross>  ?Movie_gross .
    ?Actor    <http://dbpedia.org/ontology/birthName>  ?Actor_birthName .
    ?Editor   <http://dbpedia.org/ontology/birthName>  ?Editor_birthName .
    ?Producer  <http://dbpedia.org/ontology/birthName>  ?Producer_birthName .
    ?Film_Director
              <http://dbpedia.org/ontology/birthName>  ?Film_Director_birthName .
    ?Movie    <http://dbpedia.org/ontology/budget>  ?Movie_budget ;
              <http://www.movieontology.org/2009/10/01/movieontology.owl#hasProducer>  ?Producer ;
              <http://www.movieontology.org/2009/10/01/movieontology.owl#hasDirector>  ?Film_Director ;
              <http://www.movieontology.org/2009/10/01/movieontology.owl#hasActor>  ?Actor
  }
